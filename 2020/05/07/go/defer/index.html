执行顺序

先进后出的栈，在return后倒序执行

package main

import "fmt"

func main() {
    defer func1()
    defer func2()
    defer func3()
}

func func1() {
    fmt.Println("A")
}

func func2() {
    fmt.Println("B")
}

func func3() {
    fmt.Println("C")
}
C
B
A

作用于函数返回值参数

package main

import "fmt"

func returnWithDefer() (t int) {
    // 1. t = 0
    defer func() {
        t = t * 10 // 3. t= 1*10
    }()

    return 1 // 2. t = 1
}

func main() {
    fmt.Println(returnWithDefer())
}
10

panic和recover执行顺序

根据出栈顺序先执行defer后输出panic，如不捕获panic直接异常退出

package main

import (
    "fmt"
)

func main() {
    f()

    fmt.Println("main return")
}

func f() int {
    defer func() {
        fmt.Println("defer1 before panic, with recover")
        if err := recover(); err != nil {
            fmt.Println(err)
        }
    }()

    defer func() {
        fmt.Println("defer2 before panic, without recover")
    }()

    fmt.Println("normal message before panic")

    panic("panic message")

    defer func() { fmt.Println("defer after panic is unreachable") }()

    return 0
}
normal message before panic
defer2 before panic, without recover
defer1 before panic, with recover
panic message
main return

panic捕获顺序

根据出栈顺序捕获最后一个panic

package main

import "fmt"

func main() {
    defer func() {
        if err := recover(); err != nil {
            fmt.Println(err)
        } else {
            fmt.Println("fatal")
        }
    }()

    defer func() {
        panic("defer1 panic message")
    }()

    defer func() {
        panic("defer2 panic message")
    }()

    panic("panic message")
}
defer1 panic message

package main

import "fmt"

func f(src string) (dst string) {
    fmt.Println(src)
    dst = src + "'"

    return
}

func main() {
    defer f(f("a"))
    defer f(f("b"))
}

b' // f(f("b")
a' // f(f("a")

defer

defer在编译阶段转换为runtime.deferproc，创建defer函数并注册到原函数中，使用单向链表保存，新注册defer添加到链表头
原函数的exit中插入runtime.deferreturn，负责函数exit返回前执行链表中的defer函数
per-P保存本地defer池，sched保存全局defer池，一般defer创建每次先从本地取，若本地池空，从全局池取到本地填满一半，用完后返回池子。如果是大空间defer使用mallocgc新分配空间，用完释放空间，不返回池子。

type _defer struct {
    siz     int32 // 参数+返回
    sp        uintptr  // 栈指针
    pc        uintptr  // 返回地址
    fn        *funcval // 传入函数
    _panic    *_panic  // panic that is running defer
    link      *_defer // 单向链表指针，指向下一个defer
}

func deferproc(siz int32, fn *funcval) {
    ...

    d := newdefer(siz)
    d.fn = fn
    d.pc = callerpc
    d.sp = sp
    switch siz {
    case 0:
        // Do nothing.
    case sys.PtrSize:
        *(*uintptr)(deferArgs(d)) = *(*uintptr)(unsafe.Pointer(argp))
    default:
        memmove(deferArgs(d), unsafe.Pointer(argp), uintptr(siz))
    }

    return0()
}

func newdefer(siz int32) *_defer {
    var d *_defer
    sc := deferclass(uintptr(siz))
    gp := getg()
    if sc < uintptr(len(p{}.deferpool)) {
        // 普通defer使用per-P本地池，如果为空从全局池取，取满到本地池一半的量
    }
    if d == nil {
        // Allocate new defer+args.
        systemstack(func() {
            total := roundupsize(totaldefersize(uintptr(siz)))
            d = (*_defer)(mallocgc(total, deferType, true))
        })
    }
    d.siz = siz
    d.heap = true
    d.link = gp._defer
    gp._defer = d
    return d
}

func deferreturn(arg0 uintptr) {
    gp := getg()
    d := gp._defer
    if d == nil {
        return
    }
    sp := getcallersp()
    if d.sp != sp { // 判断是否为caller注册的defer，用于函数返回
        return
    }

    if d.openDefer {
		done := runOpenDeferFrame(gp, d)
		if !done {
			throw("unfinished open-coded defers in deferreturn")
		}
		gp._defer = d.link
		freedefer(d)
		return
	}

    switch d.siz {
    case 0:
        // Do nothing.
    case sys.PtrSize:
        *(*uintptr)(unsafe.Pointer(&arg0)) = *(*uintptr)(deferArgs(d))
    default:
        memmove(unsafe.Pointer(&arg0), deferArgs(d), uintptr(d.siz))
    }
    fn := d.fn
    d.fn = nil
    gp._defer = d.link
    freedefer(d)
    _ = fn.fn
    jmpdefer(fn, uintptr(unsafe.Pointer(&arg0))) // 实现defer循环
}

func freedefer(d *_defer) {
    if !d.heap {
        return
    }
    sc := deferclass(uintptr(d.siz))
    if sc >= uintptr(len(p{}.deferpool)) { // 大空间defer不返回池子中
        return
    }
    pp := getg().m.p.ptr()
    if len(pp.deferpool[sc]) == cap(pp.deferpool[sc]) {
        // 本地per-P的defer池子满，先放一半到全局defer池
    }

    // 清空d信息

    pp.deferpool[sc] = append(pp.deferpool[sc], d) // 加入本地池
}

嵌套defer

  1. A的defer链表为A2->A1
  2. A返回前，A2.sp==sp of A，将A2从链表删除，同时执行A2的函数_ = fn.fn，
    链表变为B2->B1->A1，执行jmpdefer跳转到A2
  3. A2返回前，B2.s==sp of A2，将B2从链表删除，链表变为B1->A1
  4. A2返回前，B1.s==sp of A2，将B1从链表删除，链表变为A1
  5. 此时A1.sp!=sp of A2，退出A2 defer的deferreturn，A2执行结束
  6. A返回前，A1.sp==sp of A，将A1从链表删除
  7. A返回
func A() {
    defer A1()
    defer A2()
}

func A2() {
    defer B1()
    defer B2()
}

panic / recover

panic时，如果g有注册defer链表，使用reflectcall调用defer函数
若无recover，调用fatalpanic终止程序
若有recover，通过设置recovered字段标示，最后调用recovery恢复程序执行

type _panic struct {
  argp unsafe.Pointer // defer函数指针
  arg interface{} // panic参数
  link *_panic // panic的单向链表指针
  pc uintptr
  sp unsafe.Pointer
  recovered bool // recover标示
  aborted bool
  goexit bool
}

func gopanic(e interface{}) {
    gp := getg()

    var p _panic
    p.arg = e
    p.link = gp._panic
    gp._panic = (*_panic)(noescape(unsafe.Pointer(&p)))

    for {
        d := gp._defer
        if d == nil {
            break
        }

        // 执行reflectcall调用注册的defer函数
        if d.openDefer {
            done = runOpenDeferFrame(gp, d)
            if done && !d._panic.recovered {
                addOneOpenDeferFrame(gp, 0, nil)
            }
        } else {
            p.argp = unsafe.Pointer(getargp(0))
            reflectcall(nil, unsafe.Pointer(d.fn), deferArgs(d), uint32(d.siz), uint32(d.siz))
        }

        if p.recovered {
            gp._panic = p.link
            for gp._panic != nil && gp._panic.aborted {
                gp._panic = gp._panic.link
            }
            if gp._panic == nil { // must be done with signal
                gp.sig = 0
            }
            gp.sigcode0 = uintptr(sp)
            gp.sigcode1 = pc
            mcall(recovery) // 调用recovery恢复程序
            throw("recovery failed") // mcall should not return
        }
    }

    preprintpanics(gp._panic)

    fatalpanic(gp._panic) // 终止程序
    *(*int)(nil) = 0      // not reached
}

func gorecover(argp uintptr) interface{} {
    gp := getg()
    p := gp._panic
    if p != nil && !p.goexit && !p.recovered && argp == uintptr(p.argp) {
        p.recovered = true
        return p.arg
    }
    return nil
}

func recovery(gp *g) {
    sp := gp.sigcode0
    pc := gp.sigcode1

    gp.sched.sp = sp
    gp.sched.pc = pc
    gp.sched.lr = 0
    gp.sched.ret = 1
    gogo(&gp.sched)
}
